            function resizeIframe(obj) {
                var iframeHeight = getDocHeight(obj.contentWindow.document);

                var totalHeight = iframeHeight;
                obj.style.minHeight = iframeHeight + 'px';
            }
            function getDocHeight(D) {
                return Math.max(
                    Math.max(D.body.scrollHeight, D.documentElement.scrollHeight),
                    Math.max(D.body.offsetHeight, D.documentElement.offsetHeight),
                    Math.max(D.body.clientHeight, D.documentElement.clientHeight)
                );
            }

    function loginForm() {
        Mevzuat.Ajax("/Login/SignIn", { alert: false, form: "#UserLoginForm" }, function (e) {
            if (e.Status == 1) {//başarılı
                localStorage.setItem("token", e.Data.access_token);
                $("#UserLoginFormModal").modal("hide");
                this.Mevzuat.SetUser();
                if (window.location.hash === "#anayasa") {
                    setTimeout(function () {
                        window.location.reload();
                    }, 1000);
                }
            }
            else {
                document.getElementById("Username").value = "";
                document.getElementById("Password").value = "";
            }
        });
    }
